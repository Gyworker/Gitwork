#### 3.1.4 使用示例

#### 3.2.3 核心代码结构

```python
# -*- coding: utf-8 -*-
"""
数据类定义模块
"""

from dataclasses import dataclass, field
from typing import Optional, List, Dict, Any
from datetime import datetime

@dataclass
class TaskFilter:
    """任务筛选条件"""
    status: Optional[str] = None
    urgency: Optional[str] = None
    keyword: Optional[str] = None
    limit: int = 100

    def to_dict(self) -> Dict[str, Any]:
        """转换为字典"""
        return {
            'status': self.status,
            'urgency': self.urgency,
            'keyword': self.keyword,
            'limit': self.limit
        }

    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> 'TaskFilter':
        """从字典创建"""
        return cls(**{k: v for k, v in data.items() if k in cls.__annotations__})

@dataclass
class TaskCreateData:
    """任务创建数据"""
    task_name: str
    task_content: str = ""
    urgency: str = "中"
    respondent: str = ""
    # ... 更多字段

@dataclass
class ImportResult:
    """导入结果"""
    success: bool = False
    total_rows: int = 0
    imported_rows: int = 0
    errors: List[str] = field(default_factory=list)

    @property
    def success_rate(self) -> float:
        """成功率计算"""
        if self.total_rows == 0:
            return 0.0
        return self.imported_rows / self.total_rows * 100
```

#### 3.2.4 使用示例

```python
# 使用数据类
from utils.data_classes import TaskFilter, ImportResult

# 创建筛选条件
filter = TaskFilter(status='进行中', urgency='高', limit=50)
filter_dict = filter.to_dict()

# 创建导入结果
result = ImportResult(
    success=True,
    total_rows=100,
    imported_rows=95,
    errors=['第5行格式错误']
)
print(f"成功率: {result.success_rate:.1f}%")
```
